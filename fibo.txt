import java.util.Scanner;

public class NthFibonacciIterative {
    
    static int fibonacci(int n) {
        if (n <= 1)
            return n;
        int a = 0, b = 1, c = 0;
        for (int i = 2; i <= n; i++) {
            c = a + b;
            a = b;
            b = c;
        }
        return c;
    }
    
    static int fibonacciRe(int n) {
        if (n <= 1)
            return n;  
        return fibonacciRe(n - 1) + fibonacciRe(n - 2);
    }
    
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter the value of n: ");
        int n = sc.nextInt();
        
        System.out.println("Nth Fibonacci number (Iterative): " + fibonacci(n));
        System.out.println("Nth Fibonacci number (Recursive): " + fibonacciRe(n));
        
        sc.close();
    }
}
